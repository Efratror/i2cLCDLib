658 uint8_t LiquidCrystal::queue_find_cmd(qdata* buffer, qsize buf_len)
659 {
660  qsize cmd_size = 0;
661  qdata _data = 0;
662 
663  cmd_length = CMD_DATA_LENGTH - 1;
664 
665  while (queue_size() > 0)
666  {
667  //get 1 data
668  queue_pop(&_data);
669 
670  if (cmd_pos == 0 && _data != RETURN_TOUCH_BUTTON && _data != RETURN_TOUCH_CHECKBOX && _data != RETURN_TOUCH_SLIDER
671  && _data != RETURN_GET_EDIT && _data != RETURN_GET_PAGE && _data != RETURN_GET_CHECKBOX && _data != RETURN_GET_SLIDER
672  && _data != RETURN_TOUCH_EDIT && _data != RETURN_GET_TOUCH_EDIT)
673  continue;
674 
675  if (cmd_pos < buf_len)
676  {
677  if (cmd_pos == 0)
678  {
679  cmd_backup = _data;
680  }
681 
682  if (cmd_pos == (CMD_DATA_LENGTH - 1))
683  {
684  if (cmd_backup == RETURN_GET_EDIT || cmd_backup == RETURN_GET_TOUCH_EDIT)
685  cmd_length = CMD_DATA_LENGTH + _data - 1;
686 
687  }
688 
689  if (cmd_pos > (cmdMaxSize - 1))
690  {
691  cmd_pos = 0;
692  cmd_backup = 0;
693  }
694 
695  buffer[cmd_pos++] = _data;
696  }
697 
698  //get whole cmd data
699  if (cmd_pos > cmd_length)
700  {
701  cmd_size = cmd_pos;
702  cmd_pos = 0;
703  return cmd_size;
704  }
705  }
706 
707  return 0;
708 }
